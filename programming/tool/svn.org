#+TITLE: SVN

* 命令
** 信息

#+CAPTION: 基本信息：分支、版本号、最近一次提交等
#+begin_src sh
svn info
#+end_src

#+CAPTION: 文件修改状态
#+begin_src sh
svn status
#+end_src

#+CAPTION: 提交日志
#+begin_src sh
svn log
#+end_src

#+CAPTION: 文件差异
#+begin_src sh
svn diff <options> <path>
    -r, --revision # 一个或两个版本号：<rev>:<rev>
#+end_src

** 操作

#+CAPTION: 检出
#+begin_src sh
svn checkout <url> <path>
svn checkout <repo>/branches/<branch>/<path>
#+end_src

#+CAPTION: 提交
#+begin_src sh
svn commit <options>
    -m, --message "" #
#+end_src

#+CAPTION: 合并
#+begin_src sh
svn merge
#+end_src

#+CAPTION: 重置
#+begin_src sh
svn revert <options> <path>
    -R, --recursive # 路径为目录时递归操作
#+end_src

** 示例

#+CAPTION: 已提交后进行回滚
#+begin_src sh
svn update                      # 确认代码最新
svn log <path>                  # 确认要回滚到的版本号
svn merge -r <rev>:<rev> <path> # 回滚
svn diff <path>                 # 确认回滚结果
svn commit -m "Revert ..."      # 提交
#+end_src
